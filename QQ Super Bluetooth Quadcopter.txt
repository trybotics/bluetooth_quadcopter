#include<Servo.h>
Servo pin5,pin6,pin9,pin10;
int data=0,s=0;
void setup()
{
 Serial.begin(9600);
 pin5.attach(5);
 pin6.attach(6);
 pin9.attach(9);
 pin10.attach(10);
 pinMode(13,OUTPUT);
 pin6.writeMicroseconds(1500);
 pin9.writeMicroseconds(1500);
 pin10.writeMicroseconds(1500);
}
void loop()
{   
  if (Serial.available() > 0)
  {
  data = Serial.read();
  if(data<63)
  {
  s=1250+data*12;
  pin5.writeMicroseconds(s);
  }
  else if('A'==(char)data)
  {
   digitalWrite(13,HIGH);
   pin5.writeMicroseconds(1000);
   pin6.writeMicroseconds(1000);
   delay(1000);
   pin6.writeMicroseconds(1500);
   pin9.writeMicroseconds(1490);
   pin10.writeMicroseconds(1510);
   Serial.println("Started");
  }
  else if('S'==(char)data)
  {
   digitalWrite(13,LOW);
   pin5.writeMicroseconds(1000);
   pin6.writeMicroseconds(2000);
   Serial.println("Stoped");
  }
  else if('C'==(char)data)
  {
  sensorCalibration();
  Serial.println("Sensor Calibration Done");
  }
  else if('F'==(char)data)
  {
   digitalWrite(13,LOW);
   pin9.writeMicroseconds(1600);
   Serial.println("Forward");
  }
  else if('B'==(char)data)
  {
   digitalWrite(13,LOW);
   pin9.writeMicroseconds(1400);
   Serial.println("Backward");
  }
  else if('R'==(char)data)
  {
   digitalWrite(13,LOW);
   pin10.writeMicroseconds(1400);
   Serial.println("Right");
  }
  else if('L'==(char)data)
  {
   digitalWrite(13,LOW);
   pin10.writeMicroseconds(1600);
   Serial.println("Left");
  }
  else if('X'==(char)data)
  {
   digitalWrite(13,HIGH);
   pin6.writeMicroseconds(1500);
   pin9.writeMicroseconds(1490);
   pin10.writeMicroseconds(1510);
   Serial.println("No Change");
  }
  else if('U'==(char)data)
  {
   s+=100;
   pin5.writeMicroseconds(s);
   Serial.println("Up");
  }
  else if('D'==(char)data)
  {
   s-=100;
   pin5.writeMicroseconds(s);
   Serial.println("Down");
  }
  else if('I'==(char)data)
  {
   pin6.writeMicroseconds(1300);
   Serial.println("Clockwise");
  }
  else if('J'==(char)data)
  {
   pin6.writeMicroseconds(1700);
   Serial.println("Anti Clockwise");
  }
  }
}
void sensorCalibration()
{
   digitalWrite(13,HIGH);
   pin5.writeMicroseconds(1000);
   delay(3000);
   digitalWrite(13,LOW);
   pin5.writeMicroseconds(2000);
   pin6.writeMicroseconds(2000);
   delay(10000);
   digitalWrite(13,HIGH);
   pin5.writeMicroseconds(1000);
}
